62uint64_t get_logical_time_ticks_(int64_t uptime_ticks) {
63 uint64_t delta_ticks;
64 double slope, offset;
65
66 k_mutex_lock(&local.mutex, K_FOREVER);
67 {
68 if (uptime_ticks == -1) {
69 delta_ticks = k_uptime_ticks() - local.uptime_ref_ticks;
70 } else {
71 delta_ticks = (uint64_t)uptime_ticks - local.uptime_ref_ticks;
72 }
73
74 slope = local.curent_slope_ticks;
75 offset = local.curent_offset_ticks;
76 }
77 k_mutex_unlock(&local.mutex);
78
79 double corrected = (double)delta_ticks * slope + offset;
80
81 if (local.epoch_ref_valid) {
82 corrected += local.epoch_ref_ticks;
83 }
84
85 return (uint64_t)round(corrected);
86}

112void set_new_epoch_unix_ref(uint64_t epoch_ref_us){
113 uint64_t epoch_ref_ticks = us_to_ticks(epoch_ref_us);

115 k_mutex_lock(&local.mutex, K_FOREVER);

126uint64_t ticks_to_us_unix_time(uint64_t logical_tick) {
127 k_mutex_lock(&local.mutex, K_FOREVER);
128
129
130 int64_t delta_ticks = (int64_t)logical_tick - (int64_t)local.epoch_ref_ticks;
131 int64_t delta_us = (int64_t)ticks_to_us(delta_ticks);
132 uint64_t result = local.epoch_ref_us + delta_us;
133
134 k_mutex_unlock(&local.mutex);
135 return result;
136}

163uint64_t uncompress_time(uint32_t compress_32bit_timestamp){
164 uint64_t current_time_us = get_logical_time_us();
165
166 uint64_t uncompressed_timestamp = (current_time_us & 0xFFFFFFFF00000000ULL) |
167 ((uint64_t) compress_32bit_timestamp & 0x00000000FFFFFFFFULL);
168
169
170 if (uncompressed_timestamp > current_time_us){
171 uncompressed_timestamp -= 0x100000000ULL;
172 }
173
174 return uncompressed_timestamp;
175}
