48bs_sm_state_t transition(bluesync_role_t role, bs_sm_state_t current_state, bs_sm_event_t event) {
49 switch (current_state) {
50 case BS_SCAN_WAIT_FOR_SYNC:
51 if (event == EVENT_NEW_SYNC_RCV){
52 LOG_DBG("Transition: BS_SCAN_WAIT_FOR_SYNC -> BS_SYNC");
53 return BS_SYNC;
54 }
55 break;
56 case BS_SYNC:
57 if (event == EVENT_SYNC_EXPIRED){
58 LOG_DBG("Transition: BS_SYNC -> BS_UPDATE");
59 return BS_UPDATE;
60 }
61 break;
62 case BS_UPDATE:
63 if (event == EVENT_UPDATE_SUCCESS){
64 LOG_DBG("Transition: BS_UPDATE -> BS_ADV");
65 return BS_ADV;
66 }
67 else if (event == EVENT_UPDATE_FAILED){
68 LOG_DBG("Transition: BS_UPDATE -> BS_SCAN_WAIT_FOR_SYNC");
69 return BS_SCAN_WAIT_FOR_SYNC;
70 }
71 break;
72 case BS_ADV:
73 if (event == EVENT_ADV_EXPIRED){
74 if (role == BLUESYNC_AUTHORITY_ROLE){
75 LOG_DBG("Transition: BS_ADV -> BS_STOP");
76 return BS_STOP;
77 }
78 else if (role == BLUESYNC_CLIENT_ROLE){
79 LOG_DBG("Transition: BS_ADV -> BS_WAIT_SYNC");
80 return BS_SCAN_WAIT_FOR_SYNC;
81 }
82 }
83 break;
84 case BS_STOP:
85 if (event == EVENT_NEW_NET_SYNC){
86 LOG_DBG("Transition: BS_STOP -> BS_ADV");
87 return BS_ADV;
88 }
89 break;
90 default:
91 if (event == EVENT_INIT){
92 if (role == BLUESYNC_AUTHORITY_ROLE){
93 LOG_DBG("Transition: INIT -> BS_STOP");
94 return BS_STOP;
95 }
96 else if (role == BLUESYNC_CLIENT_ROLE){
97 LOG_DBG("Transition: INIT -> BS_WAIT_SYNC");
98 return BS_SCAN_WAIT_FOR_SYNC;
99 }
100 }
101 break;
102 }
103
104 return current_state; // No state change
105}

107void bs_state_machine_run(bs_sm_event_t event) {
108 if (sm_param.handlers == NULL){
109 LOG_ERR("Error: no handlers set...");
110 return;
111 }
112
113 k_mutex_lock(&sm_param.mutex, K_FOREVER);
114 {
115 sm_param.current_state = transition(sm_param.role, sm_param.current_state, event);
116 }
117 k_mutex_unlock(&sm_param.mutex);
118
119
120 // Call the appropriate handler for the current state
121 switch (sm_param.current_state) {
122 case BS_SCAN_WAIT_FOR_SYNC:
123 if (sm_param.handlers->bs_scan_wait_for_sync_cb){
124 sm_param.handlers->bs_scan_wait_for_sync_cb();
125 }
126 break;
127 case BS_SYNC:
128 if (sm_param.handlers->bs_sync_cb){
129 sm_param.handlers->bs_sync_cb();
130 }
131 break;
132 case BS_UPDATE:
133 if (sm_param.handlers->bs_update_cb){
134 sm_param.handlers->bs_update_cb();
135 }
136 break;
137 case BS_ADV:
138 if (sm_param.handlers->bs_adv_cb){
139 sm_param.handlers->bs_adv_cb();
140 }
141 break;
142 case BS_STOP:
143 if (sm_param.handlers->bs_stop_cb){
144 sm_param.handlers->bs_stop_cb();
145 }
146 break;
147 default:
148 break;
149 }
150}

160void bs_state_machine_set_role(bluesync_role_t role){
161 LOG_DBG("set role %d",role );
162 k_mutex_lock(&sm_param.mutex, K_FOREVER);
163 {
164
165 sm_param.role = role;
166 switch (sm_param.role)
167 {
168 case BLUESYNC_AUTHORITY_ROLE:
169 sm_param.current_state = BS_STOP;
170 break;
171
172 case BLUESYNC_CLIENT_ROLE:
173 sm_param.current_state = BS_SCAN_WAIT_FOR_SYNC;
174 break;
175
176 default:
177 sm_param.current_state = BS_NONE_STATE;
178 break;
179 }
180 }
181 k_mutex_unlock(&sm_param.mutex);
182}
